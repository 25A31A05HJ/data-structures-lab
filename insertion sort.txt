#include<iostream>
using namespace std;
int main()
{
	int i,n,temp,j,a[10];
	cin>>n;
	for(i=0;i<n;i++)
	{
	cout<<"enter number of elements";
	cin>>n;
}
	for(i=0;i<n;i++)
	{
		cout<<"enter array elements"<<i+1;
		cin>>a[i];
	}
	for(i=1;i<n;i++)
	{
		for(j=1;j>0;j--)
		{
			if(a[j]<a[j-1])
			{
				temp=a[j];
				a[j]=a[j-1];
				a[j-1]=temp;
			}
		}
		for(i=0;i<n;i++)
		{
			cout<<"sorted array data";
			cin>>a[i];
		}
	}
}
